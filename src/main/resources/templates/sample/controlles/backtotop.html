.back-to-top {
	cursor: pointer;
	position: fixed;
	bottom: 0;
	margin:6px;
	right: 16px;

<a id="back-to-top" href="#" class="btn btn-primary btn-lg back-to-top" role="button" th:title="back to top" title="Back to Top" data-toggle="tooltip" data-placement="top">
	<i class="fa fa-chevron-up" aria-hidden="true"></i>
</a>
$(window).scroll(function () {
	if ($(this).scrollTop() > 50) {
		$('#back-to-top').fadeIn();
	} else {
		$('#back-to-top').fadeOut();
	}
});
// scroll body to 0px on click
$('#back-to-top').click(function () {
	$('#back-to-top').tooltip('hide');
	$('body,html').animate({
		scrollTop: 0
	}, 800);
	return false;
});

$('#back-to-top').tooltip('show');
